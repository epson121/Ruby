def bin_to_dec(number)
        if !number.integer?
                return "not an integer"
        end
        counter = 0
        result = 0
        while number != 0
                temp = number % 10
                if temp > 1
                        return "not a binary number"
                end
                result += temp * 2 ** counter
                counter += 1
                number /= 10
        end
        return result
end

def dec_to_bin(number)
        result = ""
        if number < 0 or !number.integer?
                return "invalid input"
        end
        while number != 0
                result.insert(0, (number % 2).to_s)
                number /= 2
        end
        return result
end
