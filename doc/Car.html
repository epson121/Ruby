def self.add_make(make)
        unless @@makes.include?(make)
                @@makes << make
                @@cars[make] = 0
        end
end

def initialize(make)
        if @@makes.include?(make)
                puts "Creating a new #{make}!"
                @make = make
                @@cars[make] += 1
                @@total_count += 1
        else
                raise "No such make: #{make}."
        end
end
